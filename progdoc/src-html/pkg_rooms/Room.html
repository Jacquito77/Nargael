001package pkg_rooms; 
002import java.util.HashMap;
003import java.util.Set;
004
005import pkg_items.ItemList;
006import pkg_items.Item;
007import pkg_engine.UserInterface;
008
009/**
010 * Classe Room faisant partie intégrante du projet Zuul
011 */
012public class Room
013{
014    private String aDescription;
015    private HashMap<String, Room> aExits;
016    //     private HashMap<String, Item> aItems;
017    private NPC aNPC;
018    private NPCList aNPCList;
019    private ItemList aItemList;
020    private String aImageName;
021    //     private Item aItem;
022    private String aName;
023    private UserInterface aGUI;
024
025    /**
026     * Constructeur de la classe
027     */
028    public Room (final String pDescription, final String pImg )
029    {
030        this.aDescription = pDescription;
031        this.aExits = new HashMap<String, Room>();
032        //      aItems = new HashMap<String, Item>();
033        this.aItemList = new ItemList();
034        this.aNPCList = new NPCList();
035        
036        this.aImageName = pImg;
037
038    }
039
040    /**
041     * Petite description
042     */
043    public String getShortDescription()
044    {
045        return aDescription;
046    }
047
048    /**
049     * Retourne les sorties possibles des salles
050     */
051    public Room getExit(final String pDir)
052    {
053        return aExits.get(pDir); 
054    }
055
056    /**
057     * Retourne les sorties disponibles
058     */
059    public String getExitString ()
060    {
061        String vString="Exits: ";
062        Set<String> vKeys = aExits.keySet();
063        for(String vS : vKeys)
064        {
065            vString+= " " + vS;
066        }
067
068        return vString;
069    }
070
071    /**
072     * Initialise les sorties avec leur direction et leur arrivée
073     */
074    public void setExit (final String pDir, final Room pRoom)
075    {
076        aExits.put(pDir, pRoom);
077    }//
078
079    /**
080     * Retourne le nom de l'image
081     */
082
083    public String getImageName()
084    {
085        return aImageName;
086    }
087
088    /**
089     * Procédure ajoutant un ou des objets à une salle
090     */public void addItemRoom(final String pI, final Item pItem)
091    {
092        this.aItemList.addItem(pI, pItem);
093    }
094
095    /**
096     * Retourne l'ItemList de la Room
097     */
098    public ItemList getItemList()
099    {   
100        return this.aItemList;
101    }
102   
103    /**
104     * Fournie une description préçise des alentours
105     */
106    public String getLongDescription()
107    {
108        String vS = "You are " + this.aDescription + ".\n" + getExitString();
109        if (!aItemList.isEmpty())
110            vS +=  "\n" + this.getItemList().getItemString();
111        
112        if (!aNPCList.isEmpty())
113            vS += "\n" + this.getNPCList().getNPCString();
114            
115        return vS;
116    } // getDescription()
117
118    /**
119     * Retourne l'Item associé à la String entrée en paramètre
120     */
121    public Item getRoomItem(final String pItemName)
122    {   
123        return this.aItemList.getItem(pItemName);
124    }
125
126    /**
127     * Procédure ajoutant un ou des NPC à une salle
128     */
129    public void addNPCRoom(final String pI, final NPC pNPC)
130    {
131        this.aNPCList.addNPC(pI, pNPC);
132    }
133    
134     /**
135     * Retourne la liste de NPC
136     */
137    public NPCList getNPCList()
138    {
139        return this.aNPCList;
140    }
141    
142    /**
143     * Retourne le NPC associé à la String passée en paramètre
144     */
145    public NPC getNPCRoom(final String pNPCName)
146    {
147        return this.aNPCList.getNPC(pNPCName);
148    }
149    
150    /**
151     * Retourne la HashMap des sorties de la Room
152     */
153    public HashMap<String, Room> getExits()
154    {
155        return this.aExits;
156    }
157    
158    /**
159     * Retourne la sortie correspondante de la salle passée en paramètre
160     */
161    public String getDoor(final Room pRoom)
162    {
163        String vDoor = null;
164        for(String vS : aExits.keySet())
165        {
166            if(pRoom.equals(aExits.get(vS)));
167            vDoor = vS;
168        }
169
170        return vDoor;
171    }
172} // Room
