005/**

008 * @author Baptiste HULIN 
009 * @version (a version number or a date)
010 */
011public class TransporterRoom extends Room
012{
013    private String aSavedRoom;
014    
015    /**
016     * Constructeur
017     */
018    public TransporterRoom(final String pDescription, final String pImage)
019    {
020        super(pDescription, pImage);
021    }
022   
023     /**
024     * Retourne une Room aléatoire
025     */
026    @Override
027    public Room getExit(final String pDirection)
028    {
029        RoomRandomizer vRandom = new RoomRandomizer(GameEngine.getRooms());
030        return vRandom.nextRoom();
031    }
032    
033    /**
034     * Méthode permettant de fixer une Room pour la commande alea.
035     */
036    public void setSavedRoom(final String pString)
037    {
038        this.aSavedRoom = pString;
039    }
040
041
042}
