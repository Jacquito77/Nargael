024    /**
025     * Construit une interface utilisateur

034    /**
035     * Affiche du texte dans la zone de texte
036     */
037    public void print (String text)
038    {
039        log.append(text);
040        log.setCaretPosition(log.getDocument().getLength());
041    }
042
043    /**
044     * Affiche du texte dans la zone de texte, suivi d'un saut de ligne
045     */
046    public void println (String text)
047    {
048        log.append(text + "\n");
049        log.setCaretPosition(log.getDocument().getLength());
050    }
051
052    /**
053     * Affiche une image dans l'interface
054     */
055    public void showImage (String imageName)
056    {
057        URL imageURL = this.getClass().getClassLoader().getResource(imageName);
058        if(imageURL == null)
059            System.out.println("image not found");
060        else 
061        {
062            ImageIcon icon = new ImageIcon(imageURL);
063            image.setIcon(icon);
064            myFrame.pack();
065        }
066    }

188    /**
189     * Interface de lecteur d'action pour le champ de texte
190     */
191    public void actionPerformed(ActionEvent e)
192    {
193        
194        if(e.getActionCommand().equals("Quit"))
195            aEngine.interpretCommand("quit");
196
197        else if(e.getActionCommand().equals("Look"))
198            aEngine.interpretCommand("look");
199
200        else if(e.getActionCommand().equals("Back"))
201            aEngine.interpretCommand("back");
202
203        else if(e.getActionCommand().equals("Help"))
204            aEngine.interpretCommand("help");
205
206        else if(e.getActionCommand().equals("North"))
207            aEngine.interpretCommand("go north");
208
209        else if(e.getActionCommand().equals("East"))
210            aEngine.interpretCommand("go east");
211
212        else if(e.getActionCommand().equals("West"))
213            aEngine.interpretCommand("go west");
214
215        else if(e.getActionCommand().equals("South"))
216            aEngine.interpretCommand("go south");
217        
218        else if(e.getActionCommand().equals("Up"))
219            aEngine.interpretCommand("go up");
220        
221        else if(e.getActionCommand().equals("Down"))
222            aEngine.interpretCommand("go down");
223            
224        else processCommand();
225    }
226
227    /**
228     * Une commande a été saisie. Lire la commande et effectue le 
229     * nécessaire pour effectuer cette commande.
230     */
231    private void processCommand()
232    {
233        boolean vFinished = false;
234        String input = aEntryField.getText();
235        aEntryField.setText("");
236
237        aEngine.interpretCommand(input);
238    }
