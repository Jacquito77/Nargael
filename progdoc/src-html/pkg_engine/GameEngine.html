299    /**
300     * Procédure permettant d'exécuter la commande associée à la String entrée en paramètre
301     */
302    public static void interpretCommand(final String pCommandLine) 
303    {
304        aGUI.println(pCommandLine);
305        Command vCommand = aParser.getCommand(pCommandLine);
306
307        if (vCommand != null)
308        {
309            vCommand.execute(aPlayer);
310        }
311        else
312        {
313            aGUI.println("I don't know what you mean...");
314        }
315
316    }
