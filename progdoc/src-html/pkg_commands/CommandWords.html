001package pkg_commands; 
002import java.util.HashMap;
003 
004/**
005 * This class is part of the "World of Zuul" application. 
006 * "World of Zuul" is a very simple, text based adventure game.  
007 * 
008 * This class holds an enumeration table of all command words known to the game.
009 * It is used to recognise commands as they are typed in.
010 *
011 * @author  Michael Kolling and David J. Barnes + D.Bureau
012 * @version 2008.03.30 + 2013.09.15
013 */
014public class CommandWords
015{
016    // tableau constant qui contient tous les mots de commande valides
017    private HashMap<String, CommandWord> aValidCommands;
018    private HashMap<CommandWord, Command> aCommands;
019    
020    /**
021     * Constructeur par defaut
022     */
023    public CommandWords()
024    {
025        aValidCommands = new HashMap<String, CommandWord>();
026        aCommands = new HashMap<CommandWord, Command>();
027        for(CommandWord vCommand : CommandWord.values()) 
028        {
029            if(vCommand != CommandWord.UNKNOWN)
030            {
031                aValidCommands.put(vCommand.toString(), vCommand);
032            }
033        }
034        
035        this.aCommands.put(CommandWord.GO, new GoCommand());
036        this.aCommands.put(CommandWord.HELP , new HelpCommand());
037        this.aCommands.put(CommandWord.QUIT, new QuitCommand());
038        this.aCommands.put(CommandWord.BACK, new BackCommand());
039        this.aCommands.put(CommandWord.LOOK, new LookCommand());
040        this.aCommands.put(CommandWord.EAT, new EatCommand());
041        this.aCommands.put(CommandWord.TEST, new TestCommand());
042        this.aCommands.put(CommandWord.TAKE, new TakeCommand());
043        this.aCommands.put(CommandWord.DROP, new DropCommand());
044        this.aCommands.put(CommandWord.ITEMS, new ItemsCommand());
045        this.aCommands.put(CommandWord.TELEPORT, new TeleportCommand());
046        this.aCommands.put(CommandWord.RANDOM, new RandomCommand());
047        this.aCommands.put(CommandWord.TALK, new TalkCommand());
048        this.aCommands.put(CommandWord.SEARCH, new SearchCommand());
049        this.aCommands.put(CommandWord.GIVE, new GiveCommand()); 
050    } // CommandWords()
051
052    /**
053     * Verifie si une String donnee fait partie des commandes valides. 
054     * @param pString la String a tester
055     * @return true si pString est une comande valide, false sinon
056     */
057    public boolean isCommand( final String pString )
058    {
059       return aValidCommands.containsKey(pString);
060    } // isCommand()
061    
062    /**
063     * Permet d'obtenir la liste des commandes
064     * Retourne chaque commande supplémentaire ajoutée
065     */
066    public String getCommandList()
067    {
068        String vString = "";
069        
070        for(String command : aValidCommands.keySet())
071        {
072            vString += command + " ";
073        }
074        return vString;
075    }
076    
077    public CommandWord getCommandWord(final String pCommandW)
078    {
079        CommandWord vCommand = aValidCommands.get(pCommandW);
080        if(vCommand != null)
081        {
082            return vCommand;
083        }
084        else
085        {
086            return CommandWord.UNKNOWN;
087        }
088    }
089    
090    public Command getCommand(final CommandWord pCommandWord)
091    {
092        Command vCommand = this.aCommands.get(pCommandWord);
093        return vCommand;
094    }
095} // CommandWords
