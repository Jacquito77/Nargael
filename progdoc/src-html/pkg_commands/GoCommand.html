013/**

015 * 
016 * @author Baptiste HULIN
017 * @version (version number or date here)
018 */
019public class GoCommand extends Command
020{
021    /**
022     *  Fonction booléenne execute de la Classe qui permet d'exécuter la commande associé au nom de celle-ci
023     */
024    @Override public boolean execute(final Player pPlayer)
025    {
026        if(!this.hasSecondWord()) {  
027            
028            GameEngine.aGUI.println("Go where?");  
029            return false;  
030        }  
031
032        
033        String vDirection = this.getSecondWord();   
034        Room nextRoom = pPlayer.getCurrentRoom().getExit(vDirection);
035
036        if (nextRoom == null)  
037            GameEngine.aGUI.println("There is no door!");  
038        else {  
039            pPlayer.move(nextRoom); 
040
041            
042            int vLimit = GameEngine.getLimit();
043            vLimit = vLimit -1;
044            
045            GameEngine.setLimit(vLimit);
046            GameEngine.aGUI.println(pPlayer.getCurrentRoom().getLongDescription());  
047            if(pPlayer.getCurrentRoom().getImageName() != null)  
048                GameEngine.aGUI.showImage(pPlayer.getCurrentRoom().getImageName());  
049                
050                
051            if (vLimit == 40)
052                GameEngine.aGUI.println("THE DAY ENDS (40 steps remaining)");
053
054            if (vLimit == 20)
055                GameEngine.aGUI.println("NIGHT FALLS (20 steps remaining)");
056
057            if (vLimit == 10)
058                GameEngine.aGUI.println("IT WILL BE DARK NIGHT SOON, HURRY (10 steps remaining)");
059            if (vLimit == 0)
060            {
061                GameEngine.aGUI.println("YOU HAVE LOST.");
062                GameEngine.aGUI.enable(false);
063            }
064        }  
065
066        GameEngine.getMovingNPC().moveNPC();
067
068        return false;
069    }
070}
