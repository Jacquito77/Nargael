001package pkg_commands;
002
003import pkg_engine.Player;
004import pkg_engine.UserInterface;
005import pkg_engine.GameEngine;
006import pkg_rooms.Room;
007import pkg_items.Item; 
008import pkg_rooms.NPC; 
009
010/**
011 * Write a description of class GiveCommand here.
012 * 
013 * @author Baptiste HULIN
014 * @version (a version number or a date)
015 */
016public class GiveCommand extends Command
017{
018    /**
019     *  Fonction booléenne execute de la Classe qui permet d'exécuter la commande associé au nom de celle-ci
020     */
021    @Override public boolean execute(Player pPlayer)
022    {
023        String vItemName = this.getSecondWord (); 
024        Item vItem = pPlayer.getInventory().getAllItems().get(vItemName); 
025        
026      if (vItem == null)
027        {
028            GameEngine.aGUI.println ("You don't have this item !");
029       }
030       
031       else {
032           
033           for (NPC vNPC : pPlayer.getCurrentRoom().getNPCList().getNPCList().values()) {
034               if (vNPC.onGive(vItem)) {
035                    pPlayer.getInventory().itemTaken(vItemName);
036                    vNPC.getAllItemsPNJ().addItem(vItemName, vItem); 
037                    GameEngine.aGUI.println("You have given "+ vItemName + " to " + vNPC.getNPCName());
038                    break; 
039               }
040           }
041        }
042        
043        return false;
044    }
045}
