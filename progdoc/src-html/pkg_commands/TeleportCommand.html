001package pkg_commands; 
002
003import pkg_engine.Player;
004import pkg_engine.UserInterface;
005import pkg_engine.GameEngine;
006import pkg_items.Beamer;
007import pkg_rooms.Room;
008/**
009 * Write a description of class TeleportCommand here.
010 * 
011 * @author Baptiste HULIN 
012 * @version (a version number or a date)
013 */
014public class TeleportCommand extends Command
015{
016    /**
017     *  Fonction booléenne execute de la Classe qui permet d'exécuter la commande associé au nom de celle-ci
018     */
019    @Override public boolean execute(final Player pPlayer)
020    {
021        if (!pPlayer.getInventory().hasItem("Beamer"))
022        {
023            GameEngine.aGUI.println("You can't teleport without a Beamer.");
024            return false;
025        }
026
027        Beamer vBeamer = (Beamer) pPlayer.getInventory().getItem("Beamer");
028        if (vBeamer.isCharged())
029        {
030            Room vRoom = vBeamer.getSavedRoom();
031            if(!vRoom.equals(pPlayer.getCurrentRoom()))
032            {
033                GameEngine.aGUI.println("You are now teleporting");
034                pPlayer.move(vRoom);
035                if(pPlayer.getCurrentRoom().getImageName() != null)
036
037                    GameEngine.aGUI.showImage(pPlayer.getCurrentRoom().getImageName());
038
039                vBeamer.setCharged(false);
040            }
041            else 
042            {
043                GameEngine.aGUI.println("You are already in the saved room.");
044            }
045        }
046
047        else
048        {
049            GameEngine.aGUI.println("You saved the current room with your beamer.");
050            vBeamer.setCharged(true);
051            vBeamer.setSavedRoom(pPlayer.getCurrentRoom());
052        }
053        return false;
054    }
055}
