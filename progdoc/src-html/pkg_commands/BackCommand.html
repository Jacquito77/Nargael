008/**
009 * Write a description of class BackCommand here.
010 * 
011 * @author Baptiste HULIN 
012 * @version (a version number or a date)
013 */
014public class BackCommand extends Command
015{
016    /**
017     *  Fonction booléenne execute de la Classe qui permet d'exécuter la commande associé au nom de celle-ci
018     */
019    @Override public boolean execute(Player pPlayer)
020    {
021        Room vRoom = pPlayer.getPreviousRoom();
022        if (pPlayer.hasNoPreviousRoom() || vRoom == null)
023            GameEngine.aGUI.println("You can't go back anymore !");
024
025        else
026        {
027            //             aCurrentRoom = aStack.pop();
028            pPlayer.moveBack();
029            GameEngine.aGUI.println( pPlayer.getCurrentRoom().getLongDescription());
030            
031            String vDoor = pPlayer.getCurrentRoom().getDoor(vRoom);
032            if (vDoor == null)
033            {
034                GameEngine.aGUI.println("It's locked, you have been trapped");
035            }
036
037            if(pPlayer.getCurrentRoom().getImageName() != null)
038
039                GameEngine.aGUI.showImage(pPlayer.getCurrentRoom().getImageName());
040        
041        }
042        
043        return false;
044    }
045}
