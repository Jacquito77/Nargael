001import java.util.List;
002import java.util.ArrayList;
003
004/**
005 * Write a description of class Inventaire here.
006 * 
007 * @author (your name) 
008 * @version (a version number or a date)
009 */
010public class Inventaire
011{
012    private int aTotalWeight;
013    private List<Item> aList;
014    
015    /**
016     * Constructeur par défaut de la classe Inventaire
017     */public Inventaire()
018    {
019        aList = new ArrayList<Item>();
020        aTotalWeight = 0;
021    }
022    
023    /**
024     * Accesseur qui permet de retourner l'objet demandé en paramètre
025     */public Item getItem(final String pNom)
026    {
027        
028        for (Item pItem : aList)
029        {
030            
031            if (pItem.getItemName().equals(pNom))
032            {
033                return pItem;
034            }
035            
036        }
037       
038        return null;
039    }
040    
041    /**
042     * Fonction qui permet de savoir si l'Item demandé est bien contenu dans l'Inventaire 
043     */public boolean contientItem(final String pNom)
044    {
045        
046            if (getItem(pNom) != null)
047            {
048                return true;
049            }
050            else 
051            return false;
052            
053       
054    }
055    
056    
057    /**
058     * Procédure qui ajoute un Item dans l'Inventaire
059     */public void ajouteItem(final String pName, final int pWeight, final String pDescription)
060    {
061       Item pItem = new Item(pName, pWeight, pDescription);
062         if (contientItem(pName) == false)
063       {
064             aList.add(pItem);
065             aTotalWeight += pWeight;
066            }
067    }
068    
069    
070}
